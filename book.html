-- ============================================================ -- book_questions — أسئلة مجهولة لدرس الكتاب -- ============================================================ -- 1. الجدول create table if not exists book_questions ( id bigint generated always as identity primary key, question text not null, created_at timestamptz not null default now() ); -- RLS alter table book_questions enable row level security; -- الكل يقرأ create policy "public read book_questions" on book_questions for select using (true); -- الكل يكتب (مجهول) create policy "anyone insert book_questions" on book_questions for insert with check (true); -- الأدمن بس يحذف (role = 'admin' في جدول members) create policy "admin delete book_questions" on book_questions for delete using ( exists ( select 1 from members where id = (select (current_setting('request.jwt.claims', true)::json->>'sub')::uuid) and role = 'admin' ) ); -- ============================================================ -- 2. RPCs -- ============================================================ -- بعت سؤال مجهول create or replace function submit_book_question(p_question text) returns void language plpgsql security definer as $$ begin insert into book_questions(question) values (trim(p_question)); end;$$; -- جيب كل الأسئلة (الأحدث أولاً) create or replace function get_book_questions() returns table(id bigint, question text, created_at timestamptz) language plpgsql security definer as $$ begin return query select bq.id, bq.question, bq.created_at from book_questions bq order by bq.created_at desc; end;$$; -- الأدمن يحذف سؤال واحد create or replace function admin_delete_book_question(p_id bigint, p_admin_id uuid) returns void language plpgsql security definer as $$ begin if not exists(select 1 from members where id = p_admin_id and role = 'admin') then raise exception 'not admin'; end if; delete from book_questions where id = p_id; end;$$; -- الأدمن يمسح كل الأسئلة create or replace function admin_clear_book_questions(p_admin_id uuid) returns void language plpgsql security definer as $$ begin if not exists(select 1 from members where id = p_admin_id and role = 'admin') then raise exception 'not admin'; end if; delete from book_questions; end;$$;
